SELECT IL.LIST_NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUESLISTS IL
JOIN ISSUES I ON IL.ID = I.ISSUESLISTS_ID;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ASSIGNED_ISSUES
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO;
